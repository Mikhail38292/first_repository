public class Main
{  
public static void main(String args[])   
{  
int a=100;
int b=100;
System.out.print("Чётные числа от 1 до "+a+": ");
for (int i=1; i<=a; i++)
{
if (i%2==0)
{
System.out.print(i + " ");
}
}
System.out.print("\nНечётные числа от 1 до "+b+": ");
for (int i=1; i<=b; i++)
{
if (i%2!=0)
{
System.out.print(i + " ");
}
}
}
}
